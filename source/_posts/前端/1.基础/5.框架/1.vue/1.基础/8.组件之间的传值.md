### 1.$listeners

```
<template>
    <my-button @click="change" @mouseup='change'></my-button>
</template>
<script>

export default{
    data(){
content: '点我啊'
    },
    methods:{
                change(){
                    alert(1)
                }
            },
    component:{
        'MyButton': {//this.$attrs this.$listeners v-on绑定所有的方法
                    template: `<div>
                        <button @click='$listeners.click()'>点我啊</button>
                        <button @click="$emit('click')">点我啊</button>
                        <button v-on="$listeners">点我啊</button>
                        </div>`
                }
    }
}
</script>
```
