## package.json

```
{
  "name": "eg2",
  "version": "1.0.0",
  "description": "",
  "main": "index.js",
  "scripts": {
    "build": "webpack --config webpack.config.js",
    "dev": "webpack-dev-server --open --mode development"
  },
  "keywords": [],
  "author": "",
  "license": "ISC",
  "devDependencies": {
    "webpack": "^4.28.3",
    "webpack-cli": "^3.1.2",
    "webpack-dev-server": "^3.1.14"
  }
}
```

## webpack.config.js

```
const path = require("path");
const HtmlWebpackPlugin = require("html-webpack-plugin");
//开发服务器配置
module.exports = {
  devServer: {
    contentBase: "./dist",
    host: "localhost",
    port: 8080,
    //服务器返回给浏览器的时候是否启动gzip压缩
    compress: true
  }
};
```

## 运行

```
npm run dev
```

打开http://localhost:8080
